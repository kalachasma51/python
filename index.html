class Account:
    def __init__(self, accountno):
        self.accountno = accountno
        self.balance = 100
        
    def deposit(self, amt):  
        self.balance += amt
        
    def withdraw(self, amt):
        if self.balance - amt >= 100:
            self.balance -= amt
            
def prompt():
    print("1 New acc 2 withdraw 3 deposit4 highest balance 5 display 6 exit")
    
accountholders = []

def getAccountObj(accno):
    for ele in accountholders:
        if ele.accountno == accno:
            return ele
    return -1

while True:
    prompt()
    choice = int(input())
    
    if choice == 1:
        accno = int(input("Enter the account no"))
        Obj = Account(accno)
        accountholders.append(Obj)
    elif choice == 2:
        accno = int(input("Enter the account no"))
        Obj = getAccountObj(accno)
        if Obj == -1:
            print("Invalid input")
        else:
            amount = int(input("Enter the amount"))
            Obj.withdraw(amount)
    elif choice == 3:
        accno = int(input("Enter the account no"))
        Obj = getAccountObj(accno)
        if Obj == -1:
            print("Invalid input")
        else:
            amount = int(input("Enter the amount"))
            Obj.deposit(amount)
    elif choice == 4:
        high = 0
        highacc = ''
        for ele in accountholders:
            if ele.balance > high:
                high = ele.balance
                highacc = ele.accountno
        print("The account ", highacc, "has highest balance of:", high)

        accno = int(input("Enter the account no"))
        Obj = getAccountObj(accno)
        if Obj == -1:
            print("Invalid input")
        else:
            print("Balance:", Obj.balance)
    elif choice == 6:
        break

def fib(n):
    fibi=[0,1]
    for i in range(n):
        nextval=fibi[-1]+fibi[-2]
        fibi.append(nextval)
    return fibi[:n]
