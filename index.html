def binarsearch(list,n):
  list.sort()
  low =0
  high = len(list)-1
  mid =0

  while low<=high:
    mid = (low+high)//2

    if list[mid]n:
      high = mid-1
    else:
      return mid
  return -1

key = int(input('enter the key u want to search'))
result = binarsearch(list,key)

  print('key not found')
else:
  print('key is found in the position',result)
